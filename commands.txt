for f in *.vcf.gz; do echo $f; bcftools annotate --threads 24 -a ~/Jumis/data/grch/37/variants/9606_b150/All_20170403.vcf.gz -c ID $f | bgzip -@ 24 -c > $f.dbsnp_rsided.vcf.gz; tabix $f.dbsnp_rsided.vcf.gz; done
